Xem Dự Án Liên Hệ 

class BackendDeveloper {
    constructor() {
        this.name = "Hoàng";
        this.role = "Backend Developer";
        this.skills = [
            "Node.js", "Express", 
            "MongoDB", "SQL", 
            "API Design", "Docker"
        ];
    }

    createSolution(problem) {
        return this.skills
            .filter(skill => skill.canSolve(problem))
            .map(skill => skill.implement());
    }
}
