class JavaDeveloper {
    constructor() {
        this.name = "Hoàng";
        this.role = "Java Backend Developer";
        this.skills = [
            "Java", "Spring Boot", "RESTful APIs",
            "MySQL", "MongoDB", "Spring Security",
            "Docker", "WebSocket/STOMP"
        ];
        this.focus = "Scalable APIs & Secure Systems";
        this.achievements = [
            "Top 20-30 Database Design 2024",
            "Semi-finalist - Find IT Talent 2023, 2024"
        ];
    }

    buildSolution(problem) {
        const solution = this.skills
            .filter(skill => skill.canSolve(problem))
            .map(skill => skill.implement());
        
        return {
            isScalable: true,
            isSecure: true,
            solution: solution
        };
    }
    
    keepLearning() {
        return {
            currentFocus: "Microservices & Cloud Deployment",
            technologies: ["OAuth2", "JWT", "Docker"]
        };
    }
}
